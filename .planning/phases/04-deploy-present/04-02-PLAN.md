**Step 2 — Create global i18n swap script in BaseLayout.astro:**

Add a `<script>` tag in BaseLayout.astro (after the existing scripts) with this logic:

```typescript
function applyLanguage(lang: string) {
  document.querySelectorAll('[data-i18n-de]').forEach(el => {
    const value = el.getAttribute(`data-i18n-${lang}`);
    if (value) el.textContent = value;
  });
}

// Observe lang attribute changes on <html>
const observer = new MutationObserver((mutations) => {
  for (const m of mutations) {
    if (m.attributeName === 'lang') {
      const lang = document.documentElement.lang || 'en';
      applyLanguage(lang);
    }
  }
});
observer.observe(document.documentElement, { attributeFilter: ['lang'] });

// On page load (including View Transitions), restore saved language
function restoreLanguage() {
  const saved = localStorage.getItem('kunsthaus-lang');
  if (saved && saved !== 'en') {
    document.documentElement.setAttribute('lang', saved);
    applyLanguage(saved);
  }
}

document.addEventListener('astro:page-load', restoreLanguage);
if (document.readyState !== 'loading') restoreLanguage();
```
