Frontmatter script:
```astro
---
import BaseLayout from '../layouts/BaseLayout.astro';
import HeroCarousel from '../components/hero/HeroCarousel.astro';
import SectionWrapper from '../components/sections/SectionWrapper.astro';
import CinematicReveal from '../components/sections/CinematicReveal.astro';
import PinnedNarrative from '../components/sections/PinnedNarrative.astro';
import StandardGrid from '../components/sections/StandardGrid.astro';
import StickyCTA from '../components/ui/StickyCTA.astro';
import SectionProgress from '../components/ui/SectionProgress.tsx';
import Footer from '../components/footer/Footer.astro';
import { getCollection } from 'astro:content';

const lang = 'en';
const planningContent = await getCollection('planning');
const uiStrings = await getCollection('uiStrings');

// Extract sections by type
const hero = planningContent.find(s => s.data.section === 'hero');
const sections = planningContent
  .filter(s => s.data.section !== 'hero')
  .sort((a, b) => a.data.priority - b.data.priority);

// Helper: get ui string by key
const getString = (key: string) => uiStrings.find(s => s.data.key === key)?.data.value;

// D-29: Section to component type mapping
const sectionComponent: Record<string, string> = {
  'exhibitions': 'mixed',
  'offerings': 'pinned-narrative',
  'education': 'pinned-narrative',
  'key-info': 'standard-grid',
  'events': 'standard-grid',
  'planning-details': 'standard-grid',
};

// D-26: Background rotation - 4 step cycle
const bgSteps = [1, 2, 3, 4] as const;

// Section names for progress indicator
const sectionNames = sections.map(s => s.data.title[lang]);
---
```

Template:
```astro
<BaseLayout title="Kunsthaus Zurich" lang={lang}>
  <!-- Hero (fixed, z-index: 10) -->
  <HeroCarousel slides={hero.data.slides} lang={lang} uiStrings={uiStrings} />

  <!-- Spacer: pushes stacking sections below fixed hero -->
  <div class="hero-spacer"></div>

  <!-- Content sections with stacking scroll effect -->
  {sections.map((section, i) => {
    const bgStep = bgSteps[i % 4];
    const componentType = sectionComponent[section.data.section] || 'standard-grid';

    return (
      <SectionWrapper
        index={i}
        bgStep={bgStep}
        sectionName={section.data.title[lang]}
        sectionId={section.data.section}
      >
        {componentType === 'mixed' && (
          <>
            <CinematicReveal
              contentType="exhibition"
              data={{
                title: section.data.title,
                subtitle: section.data.subtitle,
                image: section.data.items?.[0]?.image,
                description: section.data.items?.[0]?.description,
                cta: { label: getString('explore-exhibition'), href: `/exhibitions/${section.data.items?.[0]?.id}` }
              }}
              lang={lang}
            />
            <PinnedNarrative
              contentType="exhibition"
              items={section.data.items?.slice(1) || []}
              lang={lang}
            />
          </>
        )}
        {componentType === 'pinned-narrative' && (
          <PinnedNarrative
            contentType={section.data.section}
            items={section.data.items || []}
            lang={lang}
          />
        )}
        {componentType === 'standard-grid' && (
          <StandardGrid
            title={section.data.title}
            body={section.data.body}
            items={section.data.items || []}
            lang={lang}
            columns={3}
          />
        )}
      </SectionWrapper>
    );
  })}

  <!-- Footer -->
  <Footer lang={lang} />

  <!-- Fixed UI elements (outside main content flow) -->
  <StickyCTA
    modeLabels={{
      planning: getString('mode-planning'),
      onsite: getString('mode-onsite'),
    }}
    ticketsLabel={getString('tickets')}
    lang={lang}
  />
  <SectionProgress
    client:visible
    totalSections={sections.length}
    sectionNames={sectionNames}
  />
</BaseLayout>
```

**Key details:**
- Hero is OUTSIDE the section loop -- it uses `hero-pinned` (fixed positioning)
- `hero-spacer` div pushes sections below the hero viewport
- Sections wrap each content section with incrementing z-index (20, 21, 22...) and rotating bgStep (1,2,3,4,1,2...)
- Exhibitions section uses BOTH CinematicReveal (first item) and PinnedNarrative (remaining items)
- StickyCTA and SectionProgress are fixed-position and placed outside section wrappers
- Footer is below all stacking sections (last element)
